59  void addIterationTask(Database &db, Table &table) {
60  auto begin = table.begin();
61  decltype(begin) end;
62  auto size = table.size();
63  tasksMutex.lock();
64  const int part = 10000;
65  if (size == 0) {
66  auto task = std::unique_ptr<TaskType>(new TaskType(this, &table, begin, begin));
67  auto t = task.get();
68  tasks.emplace_back(std::move(task));
69  db.addTask(t);
70  } else {
71  tasksSize = (size - 1) / part + 1;
72  while (size > 0) {
73  if (size >= part) {
74  size -= part;
75  end = begin + part;
76  } else {
77  size = 0;
78  end = table.end();
79  }
80  auto task = std::unique_ptr<TaskType>(new TaskType(this, &table, begin, end));
81  begin = end;
82  auto t = task.get();
83  tasks.emplace_back(std::move(task));
84  db.addTask(t);
85  }
86  }
87  tasksMutex.unlock();
88  }
